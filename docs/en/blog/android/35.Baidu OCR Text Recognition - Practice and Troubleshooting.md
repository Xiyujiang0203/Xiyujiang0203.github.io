# Baidu OCR Text Recognition — Practice & Troubleshooting

### 1. Use `registerForActivityResult` to open the camera and take a photo, get the real local image path via Uri, and pass it to the text recognition parser in the [OCR SDK](https://ai.baidu.com/ai-doc/OCR/1k3h7y3db). Use the Java sample request code and download the matching parser classes — some parts need modification, as explained below.

```kotlin
class CharacterActivity : AppCompatActivity(){

    private val TAG = "CharacterActivity"

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_character)
        characterBack.setOnClickListener { finish() }
        
        //call method
        characterTake.setOnClickListener {
            onTakePicture()
        }

    }

    var imageSaveUri : Uri?= null
    //open camera
    fun onTakePicture(){
        imageSaveUri = if(Build.VERSION.SDK_INT>= Build.VERSION_CODES.Q){
            Log.d(TAG, "onTakePicture: ${"hello"}")
            val values = ContentValues()
            values.put(MediaStore.MediaColumns.DISPLAY_NAME,"image_name.jpg")
            values.put(MediaStore.MediaColumns.RELATIVE_PATH, Environment.DIRECTORY_PICTURES)
            contentResolver.insert(MediaStore.Images.Media.EXTERNAL_CONTENT_URI,values)
        }else{
            FileProvider.getUriForFile(this,
                "${packageName}.fileprovider",
                File(externalCacheDir!!.absolutePath,"image_name.jpg")
            )
        }
        myActivityResultLauncher.launch(imageSaveUri)
        Log.d(TAG, "onTakePicture1: ${imageSaveUri}")

    }

    //get local image path
    private fun getImagePath():String{
        var cursor: Cursor? = null;
        try {
            val proj = arrayOf(MediaStore.Images.Media.DATA)
            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
                cursor = contentResolver.query(Uri.parse(imageSaveUri.toString()),proj,null,null,null)
                val column_index = cursor?.getColumnIndexOrThrow(MediaStore.Images.Media.DATA)
                cursor?.moveToFirst()
                return cursor?.getString(column_index!!)!!
            }
        }catch (e:Exception){
            e.printStackTrace()
        }finally {
            cursor?.close()
        }
        return null.toString()
    }

    //registerForActivityResult to open camera
    private val myActivityResultLauncher = registerForActivityResult(ActivityResultContracts.TakePicture()){ result : Boolean ->

        if (result) {
            Log.d(TAG, ": ${getImagePath()}")

            // Pass local path to GeneralBasic.generalBasic for parsing
            val data = GeneralBasic.generalBasic(getImagePath())
            Log.d(TAG, "${data}: ")
        }
    }

}
```

### 2. Calling `GeneralBasic.generalBasic(String imagePath)` — I modified it to accept a local image path and your own Token. See the official docs and [this article on obtaining a Token](https://blog.csdn.net/qq_61735602/article/details/127716705?spm=1001.2014.3001.5502).

```java
public class GeneralBasic {

    public static String generalBasic(String imagePath) {
        // Request URL
        String url = "https://aip.baidubce.com/rest/2.0/ocr/v1/general_basic";
        try {
            // Local file path
            String filePath = imagePath;
            byte[] imgData = FileUtil.readFileByBytes(filePath);
          
            String imgStr = Base64Util.encode(imgData);
     
            //These two lines not needed
//            String imgParam = URLEncoder.encode(imgStr, "UTF-8");
//            String param = "image=" + imgStr;

            // For demo only: fetch access_token each request. In production, cache token until expiry.
            String accessToken = "YOUR_ACCESS_TOKEN";

            String result = HttpUtil.post(url, accessToken, imgStr);
            System.out.println(result);
            return result;
        } catch (Exception e) {
            e.printStackTrace();
        }
        return null;
    }

}
```

### 3. Calling `HttpUtil` — watch out here. `ConnectionHttp` inside did not work for me; I replaced it with OkHttp for network requests. Tested and confirmed working.

```java

/**
 * HTTP utility class
 */
public class HttpUtil {

    public static String post(String requestUrl, String accessToken, String params)
            throws Exception {
        String contentType = "application/x-www-form-urlencoded";
        return HttpUtil.post(requestUrl, accessToken, contentType, params);
    }

    public static String post(String requestUrl, String accessToken, String contentType, String params)
            throws Exception {
        String encoding = "UTF-8";
        if (requestUrl.contains("nlp")) {
            encoding = "GBK";
        }
        return HttpUtil.post(requestUrl, accessToken, contentType, params, encoding);
    }

    public static String post(String requestUrl, String accessToken, String contentType, String params, String encoding)
            throws Exception {
        String url = requestUrl + "?access_token=" + accessToken;
        return HttpUtil.postGeneralUrl(url, contentType, params, encoding);
    }

    public static String postGeneralUrl(String generalUrl, String contentType, String params, String encoding)
            throws Exception {
        URL url = new URL(generalUrl);
        OkHttpClient client = new OkHttpClient();

        // Add parameters
        RequestBody body = new FormBody.Builder()
                .add("image",params)
                .build();

        Request request = new Request.Builder()
                .url(url)
                // Request header
                .addHeader("Content-Type",contentType)
                .post(body)
                .build();
        client.newCall(request).enqueue(new Callback() {
            @Override
            public void onFailure(Call call, IOException e) {
            }

            @Override
            public void onResponse(Call call, Response response) throws IOException {
                Log.d("hello", response.body().string());

            }
        });
        return null;
    }
}
```
