## 第2关查找-进阶(二分查找）

 <!-- image: removed (asset not available) -->

> 这题要求应该是返回查找次数，如果没有找到返回0。

```cpp
#include <stdio.h>

int find_pos(int data[] ,int n, int key)
{//在数组data中查找给定数据key，n是数组中数据元素的个数，返回值是数据元素比较的次数.
    
 /*------------begin-----------------*/
 int l,r,mid;
 int count = 0;
 l = 0,r = n-1;
 while(l<=r){
   mid = (l+r)/2;
   if(key<data[mid])r = mid-1;
   else if(key>data[mid]) l = mid+1;
   else return ++count;
   count++;
 }
 return 0;
 /*------------end-----------------*/
}

int main(void)
{
int a[]={-1,15,66,99,122,189};
int x;
scanf("%d",&x);
//此处填写代码，调用find_pos，在a数组中查找给定数据x的位置，并输出函数返回值
  /*------------begin-----------------*/
printf("%d",find_pos(a,6,x));
 /*------------end-----------------*/   
    
return 1;
}
```

```cpp
#include <stdio.h>
void compute(int data[],int n, int target)
{//在此处填写代码，在data数组中查找是否存在两个元素之和为target，并按要求输出
  /*-----------begin---------------*/ 
  for(int i = 0;i<n;i++){
    for(int j = i+1;j<n;j++){
      if(data[i]+data[j]==target)
      {
        printf("%d,%d",i,j);
        return;
      }
    }
  }
  printf("-1,-1");
  return;
 /*-----------end---------------*/ 

}

int main(void)
{
    int a[]={-3,2,5,9,15,32};
    int x ;
    scanf("%d",&x);
    compute(a,6,x);
}
```
