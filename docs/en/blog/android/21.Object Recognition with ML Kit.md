# Object Recognition with ML Kit

1. Add the dependency. [Official site](https://developers.google.com/ml-kit/vision/image-labeling/label-map)

> implementation 'com.google.mlkit:image-labeling:17.0.5'
> 2. Code implementation

```kotlin
class MainActivity : AppCompatActivity() {

    private val TAG = "MainActivity"

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        //Get Drawable from resources
        val drawable : Drawable = resources.getDrawable(R.drawable.cat)
        //Convert Drawable to bitmap
        val bitmapDrawable : BitmapDrawable = drawable as BitmapDrawable
        val bitmap : Bitmap = bitmapDrawable.bitmap
        //Call API to recognize image
        val image = InputImage.fromBitmap(bitmap,0)
        val labeler = ImageLabeling.getClient(ImageLabelerOptions.DEFAULT_OPTIONS)
        //Get and display results
        labeler.process(image).addOnSuccessListener { labels->
            Log.d(TAG, "onCreate: ${labels}")
        }
            .addOnFailureListener{ labeler -> }
    }
}
```
