步骤三：写ViewModel来管理数据

```kotlin
class MyViewModel : ViewModel() {

    //1 延迟初始化
    //val number:MutableLiveData<Int> by lazy { MutableLiveData<Int>().also { it.value = 0  } }
    //2 立刻初始化
    //val number = MutableLiveData(0)

    //3 MutableLiveData直接暴露给外部不太好
    //下划线表示一个内部的变量
    private val _number:MutableLiveData<Int> by lazy { MutableLiveData<Int>().also { it.value=0 } }
    val number:LiveData<Int>
    get() = _number
    
    fun addScore(aNumber:Int){
        //空才执行
        _number.value = _number.value?.plus(aNumber)
        //无论空不空都执行
        //number.value = number.value!!.plus(aNumber)
    }
}
```

步骤四：在活动中观察数据变化然后修改布局

```kotlin
class MainActivity : AppCompatActivity() {

    private var good:Button?=null
    private var bad:Button?=null
    private var textView:TextView?=null

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        good = findViewById(R.id.good)
        bad = findViewById(R.id.bad)
        textView = findViewById(R.id.textView)
		
        val viewModel = ViewModelProvider(this)[MyViewModel::class.java]
        viewModel.number.observe(this) { textView?.text = it.toString() }

        good?.setOnClickListener {
            viewModel.addScore(1)
        }
        bad?.setOnClickListener {
            viewModel.addScore(-1)
        }
    }

}
```
