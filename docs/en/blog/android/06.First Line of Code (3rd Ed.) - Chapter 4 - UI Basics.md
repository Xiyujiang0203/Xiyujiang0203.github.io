#### 4.5.3 Improving ListView Performance

```kotlin
class FruitAdapter(activity: Activity, val resourceId: Int,data:List<Fruit>) : ArrayAdapter<Fruit>(activity,resourceId,data){

    //called when each item scrolls into view
    //convertView caches previously loaded layout for reuse
    override fun getView(position: Int, convertView: View?, parent: ViewGroup): View {
        val view:View
        val  viewHolder: ViewHolder
        if (convertView == null) {
            view = LayoutInflater.from(context).inflate(resourceId,parent,false)
            val fruitImage : ImageView = view.findViewById(R.id.fruitImage)
            val fruitName : TextView = view.findViewById(R.id.fruitName)
            //store instance in ViewHolder, then in Tag
            viewHolder = ViewHolder(fruitImage,fruitName)
            view.tag = viewHolder
        }else{
            view = convertView
          	//retrieve from Tag when cached
            viewHolder = view.tag as ViewHolder
        }

        val fruit =  getItem(position)
        if (fruit != null) {
            viewHolder.fruitImage.setImageResource(fruit.imageId)
            viewHolder.fruitName.text = fruit.name
        }
        return view
    }

  
    inner class ViewHolder(val fruitImage: ImageView, val fruitName: TextView)
}
```

```kotlin
class MsgAdapter(val msgList:List<Msg>) : RecyclerView.Adapter<RecyclerView.ViewHolder>() {
		//two ViewHolders to cache controls in two layouts
    inner class LeftViewHolder(view: View) : RecyclerView.ViewHolder(view) {
        val leftMsg: TextView = view.findViewById(R.id.leftMsg)
    }

    inner class RightViewHolder(view: View) : RecyclerView.ViewHolder(view) {
        val rightMsg: TextView = view.findViewById(R.id.rightMsg)
    }

    
    override fun getItemViewType(position: Int): Int {
        val msg = msgList[position]
        //return message type for current position
        return msg.type
    }

    //load different layout and ViewHolder per viewType
    override fun onCreateViewHolder(parent: ViewGroup, viewType: Int) = if(viewType == Msg.TYPE_RECEIVED){
        val view =
            LayoutInflater.from(parent.context).inflate(R.layout.msg_left_item, parent, false)
        LeftViewHolder(view)
    }else{
        val view =
            LayoutInflater.from(parent.context).inflate(R.layout.msg_right_item, parent, false)
        RightViewHolder(view)
    }

    override fun onBindViewHolder(holder: RecyclerView.ViewHolder, position: Int) {
        val msg = msgList[position]
        when (holder) {
            is LeftViewHolder -> holder.leftMsg.text = msg.content
            is RightViewHolder -> holder.rightMsg.text = msg.content
            else -> throw  IllegalAccessException()
        }
    }

    override fun getItemCount() = msgList.size

}
```

### 4.8 Kotlin: Late Initialization and Sealed Classes

#### 4.8.1 Late Initialization

Optimize the code above with the `lateinit` keyword

```kotlin
class MainActivity : AppCompatActivity() , View.OnClickListener{

    private val msgList = ArrayList<Msg>()

    private lateinit var adapter:MsgAdapter

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        initMsg()
        val layoutManager = LinearLayoutManager(this)
        recycleView.layoutManager = layoutManager
        //check isInitialized to avoid re-initializing
        if (!::adapter.isInitialized) {
            adapter = MsgAdapter(msgList)
        }
        recycleView.adapter = adapter
        send.setOnClickListener(this)
    }

    private fun initMsg() {
        val msg1 = Msg("Hello guy.", Msg.TYPE_RECEIVED)
        msgList.add(msg1)
        val msg2 = Msg("Hello. Who is that?",Msg.TYPE_SEND)
        msgList.add(msg2)
```

##### 4.8.2 Sealed Classes

- Problem

```
interface Result
class Success(val msg:String) : Result
class Failure(val error: Exception) :Result

fun getResultMsg(result: Result) = when (result) {
    is Success -> result.msg
    is Failure -> result.error.message
    //required for exhaustive when; never reached. If Result gains a subclass without a branch, runtime crash
    else -> throw IllegalArgumentException()
}
```

- Solution: sealed class

```kotlin
sealed class Result
//subclass needs parentheses
class Success(val msg:String) : Result()
class Failure(val error: Exception) :Result()

//when on sealed class: compiler requires all subclasses handled; new subclass forces compile error until
//a new branch is added
fun getResultMsg(result: Result) = when (result) {
    is Success -> result.msg
    is Failure -> "Error is ${result.error.toString()}"
}
```
