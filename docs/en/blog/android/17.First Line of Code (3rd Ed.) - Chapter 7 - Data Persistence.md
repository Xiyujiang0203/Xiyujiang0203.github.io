### 7.2 File Storage

#### 7.2.1 Saving Data to a File

Store simple text or binary data

1. Add an EditText

2. Code

```
class MainActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
    }

    override fun onDestroy() {
        super.onDestroy()
        val inputText = editText.text.toString()
        save(inputText)
    }

    private fun save(inputText: String) {
        try{
            val output = openFileOutput("data", Context.MODE_PRIVATE)//MODE_PRIVATE overwrites or creates; MODE_APPEND appends
            val writer = BufferedWriter(OutputStreamWriter(output))
            writer.use { //Built-in extension: closes the outer stream after the lambda finishes
                it.write(inputText)
            }
        }catch (e:IOException){
            e.printStackTrace()
        }
    }
}
```

1. File location: `/data/data/<package name>/`

#### 7.2.2 Reading Data from a File

```
val inputText = load()
if (inputText.isNotEmpty()) {
    editText.setText(inputText)
    editText.setSelection(inputText.length)//Set cursor position
    Toast.makeText(this,"Restoring succeeded",Toast.LENGTH_SHORT).show()
}

 private fun load():String {
        val content = StringBuilder()
        try{
            val input  = openFileInput("data")
            val reader = BufferedReader(InputStreamReader(input))
            reader.use{
                reader.forEachLine { //Built-in extension: each line is passed to the lambda
                    content.append(it)
                }
            }
        }catch (e:IOException){
            e.printStackTrace()
        }
        return content.toString()
    }
```

#### 7.4.2 Upgrading the Database

1. Code

```
class MyDatabaseHelper(val context: Context, name:String, version:Int) : SQLiteOpenHelper(context,name,null,version){
    private val createBook = "create table Book("+
            " id integer primary key autoincrement,"+//id as primary key, auto-increment
            "author text,"+
            "price real,"+//floating point
            "pages integer,"+
            "name text)"
		//CREATE TABLE statement
    private val createCategory = "create table Category ("+
            "id integer primary key autoincrement,"+
            "category_name text,"+
            "category_code integer)"

    override fun onCreate(p0: SQLiteDatabase) {
        p0.execSQL(createBook)
        p0.execSQL(createCategory)//New table for fresh installs
        Toast.makeText(context,"Created succeeded",Toast.LENGTH_SHORT).show()
    }

    override fun onUpgrade(p0: SQLiteDatabase, p1: Int, p2: Int) {
    //Drop old tables
        p0.execSQL("drop table if exists Book")
        p0.execSQL("drop table if exists Category")
        //Re-run create
        onCreate(p0)
    }

}
```

#### 7.4.6 Querying Data

- Query: seven parameters

1. table — table name

2. columns — which columns

3. selection 4. selectionArgs — filter rows

4. groupBy — GROUP BY columns

5. having — filter after GROUP BY

6. orderBy — sort order; default if omitted

- Return value: Cursor

```
queryData.setOnClickListener {
    val db = dbHelper.writableDatabase
    //Query all rows in Book
    val cursor = db.query("Book", null, null, null, null, null, null)
    if (cursor.moveToFirst()) {
        do {
            val name = cursor.getString(cursor.getColumnIndex("name"))
            val author = cursor.getString(cursor.getColumnIndex("author"))
            val pages = cursor.getString(cursor.getColumnIndex("pages"))
            val price = cursor.getString(cursor.getColumnIndex("price"))
            Log.d(TAG, "book name is $name")
            Log.d(TAG, "book author is $author")
            Log.d(TAG, "book pages is $pages")
            Log.d(TAG, "book price is $price")

    }
    cursor.close()
}
```

- Version 3 — add category_id column to Book

```kotlin
class MyDatabaseHelper(val context: Context, name:String, version:Int) : SQLiteOpenHelper(context,name,null,version){
    private val createBook = "create table Book("+
            " id integer primary key autoincrement,"+//id as primary key, auto-increment
            "author text,"+
            "price real,"+//floating point
            "pages integer,"+
            "name text,"+
            "category_id integer)"

    private val createCategory = "create table Category ("+
            "id integer primary key autoincrement,"+
            "category_name text,"+
            "category_code integer)"

    //Fresh install at v2 runs onCreate
    override fun onCreate(p0: SQLiteDatabase) {
        p0.execSQL(createBook)
        p0.execSQL(createCategory)

    }

    //Upgrade from v1 to v2 creates the new table
    override fun onUpgrade(p0: SQLiteDatabase, oldVersion: Int, p2: Int) {
        if(oldVersion <=1){
            //v1 → v3: both blocks run
            p0.execSQL(createCategory)
        }
        if (oldVersion <= 2) {
            //v2 → v3: add column
            p0.execSQL("alter table Book add column category_id integer")
        }
    }

}
```

#### 7.6.2 Simplifying ContentValues

- Original approach

```
val values = ContentBalues()
values.put("name","Game of Thrones")
db.insert("Book",null,values)
```

- Note: `mapOf` and `A to B` create a `Pair`; `vararg` allows multiple `Pair` parameters

```
/**Pair is the type produced by A to B
 * Any is the base class of all types, like Object in Java
 * ? means nullable
 * vararg allows passing multiple Pair arguments
 */
fun cvOf(vararg pairs: Pair<String,Any?>):ContentValues {
    val cv = ContentValues()
    for (pair in pairs) {
        val key = pair.first
        val value = pair.second
        when (value) {
            is Int -> cv.put(key,value)
            is Long -> cv.put(key,value)
            is Short -> cv.put(key,value)
            is Float -> cv.put(key,value)
            is Double -> cv.put(key,value)
            is Boolean -> cv.put(key,value)
            is String -> cv.put(key,value)
            is Byte -> cv.put(key,value)
            is ByteArray -> cv.put(key,value)
            null -> cv.putNull(key)
        }
    }
    return cv
}
```

- Usage

```
val values = cvOf("name" to "Game of Thrones","author" to "George Martin","pages" to 720,"price" to 20.85)
db.insert("Book",null,values)
```
