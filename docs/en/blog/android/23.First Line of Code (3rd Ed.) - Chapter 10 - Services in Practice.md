### 10.1 What Is a Service?

Service: a solution for running programs in the background, used to execute long-running tasks that don't require user interaction.

#### 10.2.2 Updating the UI from a Child Thread

Note: you cannot update the UI from a child thread; it must be handled asynchronously

Example: change TextView content on button click

```
package com.android.androidthreadtest

import androidx.appcompat.app.AppCompatActivity
import android.os.Bundle
import android.os.Handler
import android.os.Message
import kotlinx.android.synthetic.main.activity_main.*
import kotlin.concurrent.thread

class MainActivity : AppCompatActivity() {

    val updateText = 1

    val handler = object : Handler(){
        override fun handleMessage(msg: Message) {
            super.handleMessage(msg)

            when (msg.what) {
                updateText -> textView.text = "Nice to meet you"
            }
        }
    }

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        changeTextBtn.setOnClickListener {
            thread {
                val msg = Message()
                msg.what = updateText
                handler.sendMessage(msg)
            }
        }
    }
}
```

#### 10.2.3 Understanding the Async Message Handling Mechanism

A child thread sends a Message to the MessageQueue; the Looper picks it up and passes it to handleMessage on the main thread for processing

#### 10.2.4 Using AsyncTask

package com.android.androidthreadtest

import android.os.AsyncTask
import android.widget.Toast
import java.lang.Exception

/**
 * First: input params for AsyncTask
 * Second: progress type if UI shows progress
 * Third: result type when task completes
 * No params passed to background here
 * Int as progress unit
 * Boolean for success/failure
 */
class DownloadTask : AsyncTask<Unit, Int, Boolean>() {

    //Called before background work; e.g. show dialog
    override fun onPreExecute() {
        super.onPreExecute()
        progressDialog.show() //show progress dialog
    }
    
    //Runs on worker thread; return result; publishProgress updates UI
    override fun doInBackground(vararg p0: Unit?) = try{
        while (true) {
            val downloadPercent = doDownload()
            publishProgress(downloadPercent)
            if (downloadPercent >= 100) {
                break
            }
        }
        true
    }catch (e:Exception){
        false
    } 

    
    //Called after return
    override fun onPostExecute(result: Boolean) {
        super.onPostExecute(result)
        //Show download result
        if (result) {
            Toast.makeText(context,"Download succeeded",Toast.LENGTH_SHORT).show()
        }else  Toast.makeText(context,"Download failed",Toast.LENGTH_SHORT).show()
    }
    //Main thread callback
    override fun onProgressUpdate(vararg values: Int?) {
        super.onProgressUpdate(*values)
        
        //Update download progress
        progressDialog.setMessage("Downloaded ${values[0]%}")
    }
}

### 10.3 Services

#### 10.3.1 Defining a Service

1. Create a new service

2. Override methods

```
class MyService : Service() {

    override fun onBind(intent: Intent): IBinder {
        TODO("Return the communication channel to the service.")
    }

    //Called on create
    override fun onCreate() {
        super.onCreate()
    }

    //Called on start:
    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
        return super.onStartCommand(intent, flags, startId)
    }

    //Called on destroy; release resources
    override fun onDestroy() {
        super.onDestroy()
    }
}
```

Note: creation only happens once on first start; subsequent starts just open it

### 10.4 Service Lifecycle

- Creation runs only once. Starting the service keeps it running but there is only one service instance; stopping once shuts it down

- stopSelf() — stop from inside the service

- onBind is the callback for bindService

- If both start and bind are used, both must be cancelled to destroy the service

### 10.5 More Service Tips

#### 10.5.1 Foreground Service

- Shows an icon in the status bar; pull down the status bar to see more details

1. Permission

```
<uses-permission android:name="android.permission.FOREGROUND_SERVICE"></uses-permission>
```

1. Create a foreground service

```
override fun onCreate() {
        super.onCreate()
        val intent = Intent(this, MainActivity::class.java)
        val pi = PendingIntent.getActivity(this,0,intent,PendingIntent.FLAG_IMMUTABLE)
        val notification = NotificationCompat.Builder(this, "my_service")
            .setContentText("This is content text")
            .setContentTitle("This is content title")
            .setSmallIcon(R.drawable.small_icon)
            .setLargeIcon(BitmapFactory.decodeResource(resources, R.drawable.large_icon))
            .setContentIntent(pi)
            .build()
        startForeground(1,notification)
    }
```

#### 10.5.2 Using IntentService (Not Recommended)

Service code runs on the main thread; handling time-consuming logic can cause ANR

The standard approach is to start a child thread in each method

```
//Called on start:
    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
        Log.d(TAG, "onStartCommand: ")
        thread{
            //business logic
            //stop service after logic completes
            stopSelf()
        }
        return super.onStartCommand(intent, flags, startId)
    }
```

- This kind of asynchronous self-stopping service is IntentService, which can be used like a regular service

```
class MyIntentService : IntentService("MyIntentService") {
    override fun onHandleIntent(p0: Intent?) {
        Log.d(TAG, "Thread id is ${Thread.currentThread().name}")
    }
    private val TAG = "MyIntentService"

    override fun onDestroy() {
        super.onDestroy()
        Log.d(TAG, "onDestroy: ")
    }
}

 startIntentServiceBtn.setOnClickListener {
            Log.d(TAG, "Thread id is ${Thread.currentThread().name}")
            val intent = Intent(this, MyIntentService::class.java)
            startService(intent)
        }
```
