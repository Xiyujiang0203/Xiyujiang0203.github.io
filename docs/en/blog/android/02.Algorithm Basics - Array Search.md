int find_pos(int data[] ,int n, int key)

## Level 2: Search — Advanced (Binary Search)

 <!-- image: removed (asset not available) -->

> The requirement here is to return the number of comparisons; return 0 if not found.

```cpp
#include <stdio.h>

int find_pos(int data[] ,int n, int key)
{//Search for key in data[]; n is element count; return value is comparison count.
    
 /*------------begin-----------------*/
 int l,r,mid;
 int count = 0;
 l = 0,r = n-1;
 while(l<=r){
   mid = (l+r)/2;
   if(key<data[mid])r = mid-1;
   else if(key>data[mid]) l = mid+1;
   else return ++count;
   count++;
 }
 return 0;
 /*------------end-----------------*/
}

int main(void)
{
int a[]={-1,15,66,99,122,189};
int x;
scanf("%d",&x);
//Fill in code: call find_pos to search for x in array a and print the return value
  /*------------begin-----------------*/
printf("%d",find_pos(a,6,x));
 /*------------end-----------------*/   
    
return 1;
}
```
