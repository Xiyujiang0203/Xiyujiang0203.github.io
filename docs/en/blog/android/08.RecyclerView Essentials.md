#### 4. Override Three Core Methods

- getItemCount: specify number of list items

- onCreateViewHolder: inflate item layout for the view holder and return it

- onBindViewHolder: bind data and events to item layout

```
public class MyAdapter extends RecyclerView.Adapter<MyAdapter.MyViewHolder> {
    Context myContext;
    List<String> data;

//assign context and data source to members
    public MyAdapter(Context myContext,List<String> data) {
        this.myContext = myContext;
        this.data = data;
    }

    @NonNull
    @Override
    public MyAdapter.MyViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
        View myView= LayoutInflater.from(myContext).inflate(R.layout.item,parent,false);//inflate item into parent, return view
        MyAdapter.MyViewHolder myViewHolder = new MyViewHolder(myView);//create ViewHolder
        return myViewHolder;
    }

//called when item scrolls to center
    @Override
    public void onBindViewHolder(@NonNull MyAdapter.MyViewHolder holder, @SuppressLint("RecyclerView") int position) {
        holder.textView.setText(data.get(position));//bind data to item
        holder.textView.setOnClickListener(new View.OnClickListener() {//click listener
            @Override
            public void onClick(View view) {
                Toast.makeText(myContext,data.get(position),Toast.LENGTH_SHORT).show();
            }
        });

    }

//return item count
    @Override
    public int getItemCount() {
        return data.size();
    }

//ViewHolder for item
    public class MyViewHolder extends RecyclerView.ViewHolder{

        View myView;
        TextView textView;

        public MyViewHolder(@NonNull View itemView) {
            super(itemView);
            myView = itemView;
            textView = (TextView) itemView.findViewById(R.id.text);

        }
    }

}
```
